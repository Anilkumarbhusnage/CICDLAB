provider "aws" { region = var.region } resource "aws_key_pair" "mykeypair" { key_name = var.key_name public_key = file(var.public_key) } # to create 2 EC2 instances resource "aws_instance" "my-machine" { # Launch 2 servers for_each = toset(var.my-servers) ami = var.ami_id key_name = var.key_name vpc_security_group_ids = [var.sg_id] instance_type = var.ins_type # Read from the list my-servers to name each server tags = { Name = each.key } provisioner "local-exec" { command = <<-EOT echo [${each.key}] >> /etc/ansible/hosts echo ${self.public_ip} >> /etc/ansible/hosts EOT } }
